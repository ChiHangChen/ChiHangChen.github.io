def _relu6(x):
    return K.relu(x, max_value=6.0)
def _hard_swish(x):
    return x * K.relu(x + 3.0, max_value=6.0) / 6.0
def _return_activation( x, nl):
    if nl == 'HS':
        x = Activation(_hard_swish)(x)
    if nl == 'RE':
        x = Activation(_relu6)(x)
    return x
def _conv_block(inputs, filters, kernel, strides, nl):
    channel_axis = 1 if K.image_data_format() == 'channels_first' else -1
    x = Conv2D(filters, kernel, padding='same', strides=strides)(inputs)
    x = BatchNormalization(axis=channel_axis)(x)
    return _return_activation(x, nl)
def _bottleneck( inputs, filters, kernel, expSize, s, squeeze, nl):
    channel_axis = 1 if K.image_data_format() == 'channels_first' else -1
    input_shape = K.int_shape(inputs)
    r = s == 1 and input_shape[3] == filters
    x = _conv_block(inputs, expSize, (1, 1), (1, 1), nl)
    x = DepthwiseConv2D(kernel, strides=(s, s), depth_multiplier=1, padding='same')(x)
    x = BatchNormalization(axis=channel_axis)(x)
    if squeeze:
        input_channels = x.shape[-1]
        x1 = GlobalAveragePooling2D()(x)
        x1 = Dense(int(input_channels))(x1)
        x1 = Dense(int(input_channels), activation='hard_sigmoid')(x1)
        x1= Reshape((1,1,input_channels))(x1)
        x = Multiply()([x, x1])
    x = _return_activation(x, nl)
    x = Conv2D(filters, (1, 1), strides=(1, 1), padding='same')(x)
    x = LeakyReLU()(x)
    x = BatchNormalization(axis=channel_axis)(x)
    if r:
        x = Add()([x, inputs])
    return x
def MobileNetV3(shape):
    inputs = Input(shape=shape)
    x = _conv_block(inputs, 16, (3, 3), strides=(2, 2), nl='HS')
    x = _bottleneck(x, 16, (3, 3), expSize=16, s=1, squeeze=False, nl='RE')
    x = _bottleneck(x, 24, (3, 3), expSize=64, s=2, squeeze=False, nl='RE')
    x = _bottleneck(x, 24, (3, 3), expSize=72, s=1, squeeze=False, nl='RE')
    x = _bottleneck(x, 40, (5, 5), expSize=72, s=2, squeeze=True, nl='RE')
    x = _bottleneck(x, 40, (5, 5), expSize=120, s=1, squeeze=True, nl='RE')
    x = _bottleneck(x, 40, (5, 5), expSize=120, s=1, squeeze=True, nl='RE')
    x = _bottleneck(x, 80, (3, 3), expSize=240, s=2, squeeze=False, nl='HS')
    x = _bottleneck(x, 80, (3, 3), expSize=200, s=1, squeeze=False, nl='HS')
    x = _bottleneck(x, 80, (3, 3), expSize=184, s=1, squeeze=False, nl='HS')
    x = _bottleneck(x, 80, (3, 3), expSize=184, s=1, squeeze=False, nl='HS')
    x = _bottleneck(x, 112, (3, 3), expSize=480, s=1, squeeze=True, nl='HS')
    x = _bottleneck(x, 112, (3, 3), expSize=672, s=1, squeeze=True, nl='HS')
    x = _bottleneck(x, 160, (5, 5), expSize=672, s=2, squeeze=True, nl='HS')
    x = _bottleneck(x, 160, (5, 5), expSize=960, s=1, squeeze=True, nl='HS')
    x = _bottleneck(x, 160, (5, 5), expSize=960, s=1, squeeze=True, nl='HS')
    x = _conv_block(x, 960, (1, 1), strides=(1, 1), nl='HS')
    x = GlobalAveragePooling2D()(x)
    x = Reshape((1, 1, 960))(x)
    x = Conv2D(1280, (1, 1), padding='same')(x)
    x = LeakyReLU()(x)
    x = BatchNormalization()(x)
    x = _return_activation(x, 'HS')
    x = Dropout(0.2)(x)
    x = Flatten()(x)
    x = Dense(32)(x)
    x = Dense(16)(x)
    x = Dense(8)(x)
    age_output = Dense(1,name='age')(x)
    gender_output = Dense(2, activation='softmax', name='gender')(x)
    emotion_output = Dense(6, activation='softmax', name='emotion')(x)
    model = Model(inputs = inputs, outputs = [age_output,gender_output,emotion_output])
    return model

split_value = [0,30,60,90,120,999999]
def get_cluster(i):
    for s in range(1,len(split_value)):
        lower = split_value[s-1]
        upper = split_value[s]
        if i>lower and i<=upper:
            return s
shape_cluster = np.array([get_cluster(o) for o in original_shape])
for i in np.unique(shape_cluster):
